<script lang="ts">
import { format } from 'd3';
import Vue from 'vue';
import type { PropType } from 'vue';
import L, { LatLngExpression } from 'leaflet';
import { APIMetaData, GeoJsonData, GeoJsonFeature, getUrlStr, Url } from './dashHelper';
import dataExplorerBar from './dataExplorerBar.vue';
import 'leaflet/dist/leaflet.css';
import 'leaflet-easyprint';

type FeatureProperty = { value: number; displayValue: string; displayName: string };
type MapFeature = GeoJsonFeature & {
  displayValues: Record<string, FeatureProperty>;
};

export default Vue.extend({
  name: 'Map',
  components: {
    dataExplorerBar,
  },
  props: {
    dataSelector: {
      type: Boolean,
      default: () => false,
    },
    zoom: {
      type: Number,
      default: () => 0,
    },
    center: {
      type: Array as unknown as PropType<LatLngExpression>,
      default: () => [0, 0],
    },
    rotation: {
      type: Number,
      default: () => 0,
    },
    height: {
      type: String,
      default: () => '800px',
    },
  },
  data() {
    return {
      leafletMap: null as any,
      omitDisplay: ['start_date', 'end_date'],
      omitColoring: [
        'stfips',
        'areatype',
        'area',
        'periodtype',
        'periodyear',
        'period',
        'name',
        'extent',
        'release',
        'indcodty',
        'indcode',
        'owners',
        'ownership',
        'permits',
        'suppress',
        'start_date',
        'end_date',
      ],
      mapFeatures: [] as MapFeature[],
      metadata: {} as APIMetaData,
      displayNames: {} as Record<string, string>,
      specialValues: this.$store.getters.getModelObj.dataExceptions,
      mapZoom: this.zoom,
      mapCenter: this.center,
      tableData: {} as Record<string, FeatureProperty>,
      _selectedFeature: {} as Record<string, FeatureProperty>,
      _selectedLayer: undefined,
      featureIsLocked: false,
      vars: this.$store.getters.getApiVars,
      colorvariable: 'clear',
      colorData: {} as { min: number; max: number },
      standardizeValue: ((x: number) => x) as (x: number) => number,
    };
  },
  computed: {
    selectedFeature: {
      get(): Record<string, FeatureProperty> {
        return this._selectedFeature;
      },
      set(feature: Record<string, FeatureProperty>) {
        this._selectedFeature = feature;
        this.featureIsLocked = false;

        this.UpdateDisplayedData();
      },
      cache: false,
    },
    UpdatedmapFeatures: {
      get(): MapFeature[] {
        return this.mapFeatures;
      },
      set(result: GeoJsonData): void {
        this.metadata = result.metadata;
        this.displayNames = result.metadata.columns.reduce(
          (a, x) => ({ ...a, [x.column_name.toLocaleLowerCase()]: x.display_name }),
          {},
        );
        const numberFormat = format(',');
        this.mapFeatures = result.features.map((feature) => {
          const properties: Record<string, FeatureProperty> = {};
          Object.keys(feature.properties).forEach((variable) => {
            // Extract the type information of the variable (if any)
            const fragments = variable.split('_');
            const type = fragments[fragments.length - 1];
            const typelessVar = fragments.slice(0, fragments.length - 1).join('_');

            // Get the numeric and the string representation of the value
            const value = feature.properties[variable];
            const displayValue =
              typeof value === 'number'
                ? value > 0
                  ? numberFormat(value)
                  : -1 - value < this.specialValues.length
                  ? this.specialValues[-1 - value]
                  : '-'
                : value;

            if (type.toLowerCase() == 'm') {
              const estimateVar = `${typelessVar}_e`;
              if (properties[estimateVar].value >= 0) {
                properties[estimateVar].displayValue += ` (±${displayValue})`;
              }
            } else {
              properties[variable] = {
                value,
                displayValue,
                displayName: this.displayNames[variable] ?? variable,
              };
            }
          });

          return {
            type: feature.type,
            geometry: feature.geometry,
            properties: feature.properties,
            displayValues: properties,
          };
        });
      },
    },
    computedAPICall(): Url {
      return this.$store.getters.getAPICall;
    },
  },
  watch: {
    async computedAPICall() {
      const rndInt = Math.floor(Math.random() * this.$store.getters.getOneLiners.length + 1) - 1;

      this.$swal({
        title: 'Loading Data',
        html: `<strong> Please Wait </strong>  <strong> ${this.$store.getters.getOneLiners[rndInt]}</strong>`,
        icon: 'info',
        width: '50rem',
        showCancelButton: false,
        showLoaderOnConfirm: true,
        allowOutsideClick: false,
      });
      // @ts-ignore error with typescript. does not recognize empty argument as undefined. will be fixed with typescript upgrade
      this.$swal.showLoading();
      const delay = (ms: number) => new Promise((res) => setTimeout(res, ms));
      await Promise.allSettled([this.loadMapData(), delay(500)]);
      this.$swal.close();
    },
  },
  async mounted() {
    await this.$nextTick();
    // @ts-ignore
    const mapElement = this.$refs.leafletMapContainer as HTMLElement;

    const baseMaps = {
      // The keys of this object is what will be shown on map layer dropdowns
      Map: L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
        attribution:
          '&copy; <a href="https://www.openstreetmap.org/copyright" target="_blank">OpenStreetMap</a> contributors',
        noWrap: true,
        bounds: [
          [90, 180],
          [-90, -180],
        ], // bounds necessary to declare if nowrap is true
      }),
      Satellite: L.tileLayer(
        'https://basemap.nationalmap.gov/arcgis/rest/services/USGSImageryOnly/MapServer/tile/{z}/{y}/{x}',
        {
          maxZoom: 20,
          noWrap: true,
          bounds: [
            [90, 180],
            [-90, -180],
          ], // bounds necessary to declare if nowrap is true
          attribution: 'Tiles courtesy of the <a href="https://usgs.gov/">U.S. Geological Survey</a>',
        },
      ),

      Topography: L.tileLayer('https://{s}.tile.opentopomap.org/{z}/{x}/{y}.png', {
        noWrap: true,
        bounds: [
          [90, 180],
          [-90, -180],
        ], // bounds necessary to declare if nowrap is true

        attribution:
          'Map data: &copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> contributors, <a href="http://viewfinderpanoramas.org">SRTM</a> | Map style: &copy; <a href="https://opentopomap.org">OpenTopoMap</a> (<a href="https://creativecommons.org/licenses/by-sa/3.0/">CC-BY-SA</a>)',
      }),

      NightMode: L.tileLayer(
        'https://map1.vis.earthdata.nasa.gov/wmts-webmerc/VIIRS_CityLights_2012/default//GoogleMapsCompatible_Level{maxZoom}/{z}/{y}/{x}.jpg',
        {
          attribution:
            'Imagery provided by services from the Global Imagery Browse Services (GIBS), operated by the NASA/GSFC/Earth Science Data and Information System (<a href="https://earthdata.nasa.gov">ESDIS</a>) with funding provided by NASA/HQ.',
          noWrap: true,
          bounds: [
            [90, 180],
            [-90, -180],
          ], // bounds necessary to declare if nowrap is true
          minZoom: 1,
          maxZoom: 8,
        },
      ),
      CyclOSM: L.tileLayer('https://{s}.tile-cyclosm.openstreetmap.fr/cyclosm/{z}/{x}/{y}.png', {
        attribution: '&copy; OpenStreetMap contributors, style by CyclOSM, tiles hosted by OSM France',
        subdomains: 'abc',
        maxZoom: 20,
        noWrap: true,
        bounds: [
          [90, 180],
          [-90, -180],
        ],
      }),
      Railways: L.tileLayer('https://{s}.tiles.openrailwaymap.org/standard/{z}/{x}/{y}.png', {
        subdomains: 'abc',
        maxZoom: 19,
        opacity: 0.9,
        attribution: '© OpenRailwayMap, © OpenStreetMap contributors',
      }),
    };

    this.leafletMap = L.map(mapElement, {
      zoomSnap: 0.25,
      zoomDelta: 0.25,
      zoom: this.zoom,
      minZoom: 2,
      maxBounds: [
        [90, 180],
        [-90, -180],
      ], // northeast,southwest
      layers: [baseMaps.Map], // this is the default map view.
    }).setView(this.mapCenter as LatLngExpression, this.mapZoom);

    L.control.layers(baseMaps).addTo(this.leafletMap); // add other map selections to map

    // @ts-ignore
    this.printPlugin = L.easyPrint({
      title: 'Download Map',
      position: 'topright',
      sizeModes: ['Current'],
      exportOnly: true,
    }).addTo(this.leafletMap);
  },
  methods: {
    UpdateDisplayedData() {
      // Check if there is an object in the selectedFeature and initial...
      if (Object.keys(this.selectedFeature).length <= 0) {
        // if nothing selected show the first data from the total set
        const firstFeature = (this.UpdatedmapFeatures as MapFeature[])[0];
        if (firstFeature) this.tableData = firstFeature.displayValues;
      } else {
        this.tableData = this.selectedFeature;
      }
    },
    ShowSelectedFeature() {
      this.UpdateDisplayedData();
      this.leafletMap.eachLayer((layer: any) => {
        if (layer._url == null) this.leafletMap.removeLayer(layer);
      }); // layers with _url are base image tiles
      // @ts-ignore
      L.geoJSON(this.UpdatedmapFeatures as GeoJsonData, {
        style: this.ColorMap,
        onEachFeature: this.SelectedFeatureFn,
      }).addTo(this.leafletMap);
    },
    async showErrorDialog() {
      await this.$swal({
        // show data not found box if no data and not initial load.
        title: 'Data Not Found',
        html: '<br>Please contact us at bber@unm.edu or Live Chat Below with your Data Request <br><br> <strong>Thank You</strong>',
        icon: 'error',
        width: '50rem',
      }).then();
    },
    async loadMapData() {
      try {
        const d = await fetch(getUrlStr(this.$store.getters.getAPICall));
        if (d.status !== 200) return this.showErrorDialog();

        const result: GeoJsonData = await d.json();

        if (!result.features || Object.keys(result.features).length < 1 || !result.metadata)
          return this.showErrorDialog();

        this.UpdatedmapFeatures = result;
        this.selectedFeature = {};
        this.tableData = {};

        this.CalculateColorBounds(this.colorvariable);

        if (this.vars.selectedAreatype.toLowerCase() === 'us' || this.vars.selectedAreatype.toLowerCase() === 'state') {
          this.mapZoom = 3.25;
          this.mapCenter = [-108.14, 42.47];
        } else {
          this.mapZoom = 7.35;
          this.mapCenter = [-106, 34.35];
        }
      } catch (err) {}
    },

    async CalculateColorBounds(variable: string) {
      this.colorvariable = variable;
      const values: number[] = [];
      for (const f of this.UpdatedmapFeatures as MapFeature[]) {
        const value = f.displayValues[variable]?.value;
        if (value && value >= 0) {
          values.push(value);
        }
      }
      if (values.length == 0) {
        this.standardizeValue = (x: number) => x;
        this.colorData = { min: 0, max: 0 };
        this.ShowSelectedFeature();
        return;
      }
      // Standardize value
      if (values.length > 2) {
        // Calculate the median
        const sorted = values.sort((a, b) => a - b);
        const half = Math.floor(sorted.length / 2);
        const median = sorted.length % 2 ? sorted[half] : (sorted[half] + values[half + 1]) / 2;

        // Use a sigmoid function to calculate a color gradient
        // 90% of the values should be represented from the bound 0.2 to 0.8
        const upperBound = sorted[Math.floor(sorted.length * 0.95)];
        const sigmoidFactor = -(Math.log(Math.abs(1 / 0.9 - 1)) / upperBound);

        // The sigmoid function
        this.standardizeValue = (x: number) => {
          const x2 = 2 / (1 + Math.E ** (-sigmoidFactor * (x - median))) - 1;
          return x2;
        };
        this.colorData = {
          max: Math.max(...values.map(this.standardizeValue)),
          min: Math.min(...values.map(this.standardizeValue)),
        };
      } else {
        this.standardizeValue = (x: number) => x;
        this.colorData = {
          max: Math.max(...values),
          min: Math.min(...values),
        };
      }

      // Update the map with the new colors
      this.ShowSelectedFeature();
    },

    ColorMap(feature: MapFeature) {
      if (
        !this.colorvariable ||
        this.colorvariable === '' ||
        this.colorvariable === 'clear' ||
        this.mapFeatures.length == 1
      ) {
        return {
          fillColor: '#abd8de',
          color: '#000000',
          weight: 0.5,
          fillOpacity: 0.3,
        };
      }

      if (!feature.displayValues[this.colorvariable]) {
        this.colorvariable = 'clear';
        this.ColorMap(feature);
        return;
      }

      const value = this.standardizeValue(feature.displayValues[this.colorvariable].value);
      let rScale;
      let gScale;
      let bScale;
      if (value > 0) {
        // make the positive values go towards blue
        rScale = 250 - Math.floor(value * 200);
        gScale = 250 - Math.floor(value * 175);
        bScale = 250 - Math.floor(value * 100);
      } else {
        // make the negative values go towards red
        rScale = 250 + Math.floor(value * 10);
        gScale = 250 + Math.floor(value * 120);
        bScale = 250 + Math.floor(value * 170);
      }
      const fillColor = `#${rScale.toString(16)}${gScale.toString(16)}${bScale.toString(16)}`;
      return {
        fillColor,
        color: '#000000', // border color
        weight: 1,
        fillOpacity: 0.75,
      };
    },

    /** Update the style to a selected layer */
    SelectLayer(layer: any) {
      layer.setStyle({ weight: 4 });

      // Unselect any previous selection
      if (this._selectedLayer && this._selectedLayer != layer) {
        this.UnselectLayer(this._selectedLayer);
      }
      this._selectedLayer = layer;
    },
    /** Update the style to an unselected layer */
    UnselectLayer(layer: any) {
      layer.setStyle({ weight: 1 });

      // Remove this layer as selected (but only if it's the selected one)
      if (this._selectedLayer == layer) {
        this._selectedLayer = undefined;
      }
    },

    /** Adds the event handlers to all the layers/features on the map */
    SelectedFeatureFn(feature: MapFeature, layer: L.Layer) {
      // If a layer/feature is clicked, select it and mark the selection as locked, only another click can change the selection
      layer.on('click', () => {
        if (this.selectedFeature == feature.displayValues && this.featureIsLocked) {
          this.featureIsLocked = false;
          this._selectedLayer = undefined;
          this.UnselectLayer(layer);
        } else {
          this.SelectLayer(layer);
          this.selectedFeature = feature.displayValues;
          this.featureIsLocked = true;
        }
      });

      // If it is a hover, select the feature until..
      layer.on('mouseover', () => {
        if (this.featureIsLocked) return;
        this.SelectLayer(layer);
        this.selectedFeature = feature.displayValues;
      });
      // ...the mouse hovers out of the feature again
      layer.on('mouseout', () => {
        if (this.featureIsLocked) return;
        this.UnselectLayer(layer);
      });

      // Check once on initialization if it's already selected
      if (this.selectedFeature == feature.displayValues) {
        this.SelectLayer(layer);
      }
    },
    getVariableDescription(variable: string) {
      const description = this.metadata.columns.find((v) => v.column_name == variable)?.column_description ?? '';

      if (description.includes('!!')) {
        const lines = ['Census categories:', ...description.split('!!')];
        return lines.join('<br /><span>&#8627;</span>');
      }

      return description;
    },
  },
});
</script>

/// Start XML producer function
export async function buildXmlForRGIS(
  this: any,
  data: Record<string, any>[],
  metadata: APIMetaData,
): Promise<{ xml: Document }> {
  const connectSentences = (...sentences: string[]) => {
    let result = '';
    for (let sentence of sentences) {
      sentence = sentence.trim();
      if (sentence.slice(-1) != '.') {
        sentence += '.';
      }
      result += `${sentence} `;
    }
    return result.trim();
  };

  const parseDate = (date: string) => date.split('T')[0].split('-').join('');

  const tableData = {
    tableTitle: this.$store.getters.getApiVars.selectedData,
    tableDescription: connectSentences(metadata.table.data_title, metadata.table.data_description),
    tablePurpose: metadata.table.purpose,
    publishDate: '20190909', // TODO: This info has to be added
    creationDate: '20180909', // TODO: This data is provided in the tablsrce table: add query
    metainfoDate: parseDate(metadata.table.updated),
    dataStatus: 'Complete', // TODO: This info has to be added
    updatePeriod: this.$store.getters.getApiVars.selectedPeriodtype,
    updateType: metadata.table.time_series,
    areaTypes: this.$store.getters.getApiVars.areatype,
    pointVectorCnt: data.length,
    dataSource: metadata.table.source ?? 'UNM Bureau of Business and Economic Research',
  };

  // Get the bounding coordinates
  let west = 0;
  let east = 0;
  let north = 0;
  let south = 0;
  if (data.length > 0) {
    for (let i = 0; i < data.length; i += 1) {
      const geog = data[i].properties.extent.substring(4, data[i].properties.extent.length - 1).split(/[\s,]+/);
      if (Number(geog[0]) < west) west = Number(geog[0]);
      if (Number(geog[1]) > east) south = Number(geog[1]);
      if (Number(geog[2]) < south) east = Number(geog[2]);
      if (Number(geog[3]) > north) north = Number(geog[3]);
    }
  }

  // load template
  const res = await fetch('/RGIS_XML/template.xml');
  let d = await res.text();

  // remove all comments and convert to xml object
  d = d.replace(/<!--.*?-->/gs, '');
  const parser = new DOMParser();
  const xml = parser.parseFromString(d, 'text/xml');

  // Need to find a way to get metadata (data release date, data published date from API)
  // if (nameRecords.length > 0 && nameRecords[0].properties.releasedate !== null) {
  //     var relDate = new Date(nameRecords[0].properties.releasedate);
  //     let month = relDate.getMonth() + 1;
  //     if (month < 10) { month = "0" + month; }
  //     xml.getElementsByTagName("pubdate")[0].childNodes[0].nodeValue = relDate.getFullYear().toString() + month;
  //     filename = filename + "_" + relDate.getFullYear().toString();
  // }
  // if (nameRecords.length > 0 && nameRecords[0].properties.updated !== null) {
  //     var updatedate = new Date(nameRecords[0].properties.updated);
  //     let month = updatedate.getMonth() + 1;
  //     if (month < 10) { month = "0" + month; }
  //     xml.getElementsByTagName("caldate")[0].childNodes[0].nodeValue = updatedate.getFullYear().toString() + month;
  // }

  // helper function to escape a XML/HTML text
  const escapeHTML = (text: string) => {
    if (typeof text === 'string') {
      return text
        .replace(/&/g, '&amp;')
        .replace(/</g, '&lt;')
        .replace(/>/g, '&gt;')
        .replace(/"/g, '&quot;')
        .replace(/'/g, '&apos;');
    }
    return text;
  };

  // Define all xml tags and their corresponding values
  const xmlVariables = {
    pubdate: tableData.publishDate,
    title: tableData.tableTitle,
    abstract: tableData.tableDescription,
    purpose: tableData.tablePurpose,
    caldate: tableData.creationDate,
    progress: tableData.dataStatus,
    update: tableData.updateType,
    westbc: west,
    eastbc: east,
    southbc: south,
    northbc: north,
    indspref: tableData.tableTitle,
    ptvctcnt: tableData.pointVectorCnt,
    enttypl: tableData.tableTitle,
    enttypds: tableData.dataSource,
    metd: tableData.metainfoDate,
  };

  // Set the XML values
  const tags = Object.keys(xmlVariables);
  for (const tag of tags) {
    xml.getElementsByTagName(tag)[0].childNodes[0].nodeValue = escapeHTML(
      xmlVariables[tag as keyof typeof xmlVariables],
    );
  }

  // A helper function for all the area type entries that have to be created
  const getAreaTypeEntry = (id: string, name: string, source: string) => {
    const edom = xml.createElement('edom');
    const edomv = xml.createElement('edomv');
    const edomvd = xml.createElement('edomvd');
    const edomvds = xml.createElement('edomvds');

    edomv.innerHTML = escapeHTML(id);
    edomvd.innerHTML = escapeHTML(name);
    edomvds.innerHTML = escapeHTML(source);

    edom.appendChild(edomv);
    edom.appendChild(edomvd);
    edom.appendChild(edomvds);

    return edom;
  };

  // The three different area types that have to be inserted
  const geographyDomV = xml.createElement('attrdomv');
  const geographyNames = xml.createElement('attrdomv');
  const constantDomV = xml.createElement('attrdomv');

  // Add all area types from the result
  for (let i = 0; i < data.length; i += 1) {
    const edom = getAreaTypeEntry(data[i].properties.area, data[i].properties.name, tableData.dataSource);
    geographyDomV.appendChild(edom);
    const udom = xml.createElement('udom');
    udom.innerHTML = escapeHTML(data[i].properties.name);
    geographyNames.appendChild(udom);
  }

  // Add table constant the area types
  const areaTypeNames = Object.keys(tableData.areaTypes);
  for (let i = 0; i < areaTypeNames.length; i++) {
    const areaType = areaTypeNames[i];
    const code = tableData.areaTypes[areaType];
    if (parseInt(code) > 4) {
      const edom = getAreaTypeEntry(code, areaType, tableData.dataSource);
      constantDomV.appendChild(edom);
    }
  }

  // Insert the three area type lists into the xml
  xml.getElementsByTagName('attrdomv')[1].replaceWith(geographyDomV);
  xml.getElementsByTagName('attrdomv')[2].replaceWith(geographyNames);
  xml.getElementsByTagName('attrdomv')[3].replaceWith(constantDomV);

  // Add all attribute descriptions
  const columnDetails = metadata.columns.reduce<Record<string, { displayname: string; description: string }>>(
    (a, x) => ({ ...a, [x.column_name]: { displayname: x.display_name, description: x.column_description } }),
    {},
  );
  if (data.length > 0) {
    const xmlAttrList = xml.getElementsByTagName('detailed')[0];
    const properties = Object.keys(data[0].properties);
    let overviewText = '';

    for (const property of properties) {
      const displayName = columnDetails[property]?.displayname;
      const description = columnDetails[property]?.description;
      if (displayName != null && description != null) {
        const infoText = `${displayName}: ${description}`;

        const attr = xml.createElement('attr');
        const attrlabl = xml.createElement('attrlabl');
        const attrdef = xml.createElement('attrdef');
        const attrdefs = xml.createElement('attrdefs');
        const attrdomv = xml.createElement('attrdomv');
        const udom = xml.createElement('udom');

        attrlabl.innerHTML = escapeHTML(property);
        attrdef.innerHTML = escapeHTML(infoText);
        attrdefs.innerHTML = escapeHTML(tableData.dataSource);
        udom.innerHTML = escapeHTML(infoText);

        attrdomv.appendChild(udom);
        attr.appendChild(attrlabl);
        attr.appendChild(attrdef);
        attr.appendChild(attrdefs);
        attr.appendChild(attrdomv);

        xmlAttrList.appendChild(attr);

        overviewText += `${property} - ${displayName}\n`;
      } else {
        console.log('Missing description for the property: ', property);
      }
    }

    xml.getElementsByTagName('eaover')[0].childNodes[0].nodeValue = escapeHTML(overviewText);
  }

  // Set the table theme
  const geographyTheme = xml.createElement('themekey');
  geographyTheme.innerHTML = escapeHTML(this.$store.getters.getApiVars.selectedAreatype);
  xml.getElementsByTagName('theme')[1].appendChild(geographyTheme);

  // Add the area type column header
  const edom = getAreaTypeEntry(
    this.$store.getters.getApiVars.selectedAreatype,
    this.$store.getters.getApiVars.selectedAreatype,
    tableData.dataSource,
  );
  xml.getElementsByTagName('attrdomv')[0].appendChild(edom);

  let feature: Record<string, any> = {};
  if (data.length > 0) {
    feature = { ...data[0] };
    const edom = getAreaTypeEntry(
      feature.properties.periodtype,
      this.$store.getters.getApiVars.selectedPeriodtype,
      'UNM BBER',
    );
    xml.getElementsByTagName('attrdomv')[3].appendChild(edom);

    // Add the period year
    const udom = xml.createElement('udom');
    udom.innerHTML = escapeHTML(feature.properties.periodyear);
    xml.getElementsByTagName('attrdomv')[4].appendChild(udom);
  }

  const formatedXML = new DOMParser().parseFromString(
    formatXml(new XMLSerializer().serializeToString(xml)),
    'text/xml',
  );
  return { xml: formatedXML };
}
