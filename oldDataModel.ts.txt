export const state = () => {
    const defaultDataDisplay = Object.keys(dataExplorerOrderCSW)[0] as keyof typeof dataExplorerOrder; // USE CSW as it is the smaller of the the data. the data only used /external/cswdemo
    const _user = localStorage.getItem('user');
    const user = _user != undefined ? JSON.parse(_user) : undefined;
    return {
        DATA_CATEGORIES,
        baseAPIURL: 'https://api.bber.unm.edu/api/data/rest/' as string,
        baseCMSURL: 'https://api.bber.unm.edu/api/' as string,
        baseFilesURL: 'https://api.bber.unm.edu/' as string,
        modelObj: { countyList, placeList, dataExplorerOrder, searchCategories, dataExceptions, dataExplorerOrderCSW },
        user: {
            jwt: user?.jwt,
            user: user?.user,
        } as {
            jwt: string | undefined;
            user:
            | { username: string; email: string; products: { name: string; slug: string }[];[key: string]: any }
            | undefined;
        },
        APICall: { base: 'initial' } as Url,
        oneLiners,
        periodtypes,
        industries,
        pages: Pages,
        apiVars: {
            stfips: {} as Record<string, any>,
            areatype: {} as Record<string, any>,
            area: {} as Record<string, any>,
            periodtype: {} as Record<string, any>,
            periodyear: {} as Record<string, any>,
            period: {} as Record<string, any>,
            indcode: {} as Record<string, any>,
            adjusted: {} as Record<string, any>,
            ownership: {} as Record<string, any>,
            selectedStfips: '' as string,
            selectedAreatype: '' as string,
            selectedArea: '' as string,
            selectedPeriodtype: '' as string,
            selectedPeriodyear: '' as string,
            selectedPeriod: '' as string,
            selectedIndcode: '' as string,
            selectedAdjusted: '' as string,
            selectedOwnership: '' as string,
            selectedDataCategory: 'All' as string,
            dataList: Object.keys(dataExplorerOrder) as string[],
            dataListCSW: Object.keys(dataExplorerOrderCSW) as string[],
            dataCategories: Object.values(DATA_CATEGORIES),
            dataExplorerOrder: dataExplorerOrder,
            selectedData: defaultDataDisplay,
            api: '' as string,
            tableName: dataExplorerOrder[defaultDataDisplay].tableName as string,
        },
    };
};

export type RootState = ReturnType<typeof state>;

export const getters = {
    baseAPIURL: (s: RootState): RootState['baseAPIURL'] => s.baseAPIURL,
    baseCMSURL: (s: RootState): RootState['baseCMSURL'] => s.baseCMSURL,
    baseFilesURL: (s: RootState): RootState['baseFilesURL'] => s.baseFilesURL,
    getUser: (s: RootState): RootState['user'] => s.user,
    getModelObj: (s: RootState): RootState['modelObj'] => s.modelObj,
    getApiVars: (s: RootState): RootState['apiVars'] => s.apiVars,
    getAPICall: (s: RootState): RootState['APICall'] => s.APICall,
    getOneLiners: (s: RootState): RootState['oneLiners'] => s.oneLiners,
    getPeriodtypes: (s: RootState): RootState['periodtypes'] => s.periodtypes,
    getIndustries: (s: RootState): RootState['industries'] => s.industries,
    getPages: (s: RootState): RootState['pages'] => s.pages,
    getDATA_CATEGORIES: (s: RootState): RootState['DATA_CATEGORIES'] => s.DATA_CATEGORIES,
};

export type GettersType = { [K in keyof typeof getters]: ReturnType<(typeof getters)[K]> };

export const mutations: MutationTree<RootState> = {
    updateAPICall: (paramState: RootState, d: Url): void => {
        paramState.APICall = d;
    },
    updateAPIVars: (paramState: RootState, d: Record<string, any>[]): void => {
        paramState.apiVars.stfips = {};
        paramState.apiVars.areatype = {};
        paramState.apiVars.periodtype = {};
        paramState.apiVars.periodyear = { All: '' };
        paramState.apiVars.area = { All: '' };
        paramState.apiVars.period = { All: '' };
        paramState.apiVars.indcode = { All: '' };
        paramState.apiVars.adjusted = { All: '' };
        paramState.apiVars.ownership = { All: '' };

        d.forEach((element) => {
            if ('stfips' in element) paramState.apiVars.stfips[element.displayname] = element.stfips;

            if ('areatype' in element) paramState.apiVars.areatype[element.displayname] = element.areatype;

            if ('area' in element) paramState.apiVars.area[element.displayname] = element.area;

            if ('periodtype' in element) paramState.apiVars.periodtype[element.displayname] = element.periodtype;

            if ('periodyear' in element) paramState.apiVars.periodyear[element.periodyear] = element.periodyear;

            if ('period' in element) paramState.apiVars.period[element.period] = element.period;

            if ('indcode' in element) paramState.apiVars.indcode[element.displayname] = element.indcode;

            if ('adjusted' in element) paramState.apiVars.adjusted[element.adjusted] = element.adjusted;

            if ('ownership' in element) paramState.apiVars.ownership[element.displayname] = element.ownership;

            if ('dataList' in element) paramState.apiVars.dataList = element.dataList;

            if ('selectedArea' in element) paramState.apiVars.selectedArea = element.selectedArea;

            if ('tableName' in element) paramState.apiVars.tableName = element.tableName;
        });
        paramState.apiVars.selectedStfips = Object.keys(paramState.apiVars.stfips)[0];
        paramState.apiVars.selectedAreatype = Object.keys(paramState.apiVars.areatype)[0];
        paramState.apiVars.selectedArea = Object.keys(paramState.apiVars.area)[0];
        const periodTypes = Object.entries(paramState.apiVars.periodtype);
        periodTypes.sort((a, b) => a[1] - b[1]); // reverse sort based on areatype and use largest value. larger periodtypes represent smaller times.
        paramState.apiVars.selectedPeriodtype = periodTypes.map((entry) => entry[0]).reverse()[0];
        paramState.apiVars.selectedPeriodyear = Object.keys(paramState.apiVars.periodyear).sort().reverse()[1] ?? 'All';
        paramState.apiVars.selectedPeriod = Object.keys(paramState.apiVars.period)[0];
        paramState.apiVars.selectedIndcode = Object.keys(paramState.apiVars.indcode)[1];
        paramState.apiVars.selectedAdjusted = Object.keys(paramState.apiVars.adjusted)[0];
        paramState.apiVars.selectedOwnership = Object.keys(paramState.apiVars.ownership)[1];
    },
    updateSelectedApiVars: (
        paramState: RootState,
        updateObj: { field: keyof RootState['apiVars']; value: any },
    ): void => {
        paramState.apiVars[updateObj.field] = updateObj.value;
    },
    updateUser: (paramState: RootState, user: RootState['user'] | undefined): void => {
        if (user) {
            localStorage.setItem('user', JSON.stringify(user));
            paramState.user.jwt = user.jwt;
            paramState.user.user = user.user ?? paramState.user.user;
        } else {
            localStorage.removeItem('user');
            paramState.user.jwt = undefined;
            paramState.user.user = undefined;
        }
    },
};
